\begin{tikzpicture}
    
  \newcommand{\n}{SUB_NUM_VERTICES}
  \newcommand{\rad}{SUB_RADIUS}
  \newcommand{\labelOffset}{SUB_LABEL_OFFSET}
  \newcommand{\vertexRadius}{SUB_VERTEX_RADIUS}
  \newcommand{\start}{90+360/\n/2}
  \newcommand{\vertexAngle}[1]{{\start+360/\n*(#1-1)}}
  
  % Draw vertices
  \foreach \i in {1,...,\n} {
  \draw[fill] (\vertexAngle{\i}:\rad) circle (\vertexRadius) node at (\vertexAngle{\i}:\rad+\labelOffset) {\i};
  }

  % Draw boundary
  \draw 
    (\vertexAngle{1}:\rad)
    \foreach \i in {2,...,\n} {-- (\vertexAngle{\i}:\rad)}
    -- cycle;

  % Draw chords
  \foreach \src/\tar in {SUB_CHORDS} {
    \draw[red] (\vertexAngle{\src}:\rad) -- (\vertexAngle{\tar}:\rad);
  }
\end{tikzpicture}
